Reduce whitespace in printed lists

# create a list
alist <- list(a=1,b=2,c=list(d=1:3, e="hello"))
# regular print
alist
## $a
## [1] 1
## 
## $b
## [1] 2
## 
## $c
## $c$d
## [1] 1 2 3
## 
## $c$e
## [1] "hello"
# shorter print
print.simple.list(alist)
##      _    
## a    1    
## b    2    
## c.d1 1    
## c.d2 2    
## c.d3 3    
## c.e  hello
